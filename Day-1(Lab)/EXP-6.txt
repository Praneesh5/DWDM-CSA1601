# Program 6

age <- c(23,23,27,27,39,41,47,49,50,
52,54,54,56,57,58,58,60,61)

value <- 35

# Min-Max

minmax <- (value-min(age))/(max(age)-min(age))

cat("Min-Max =",minmax,"\n")

# Z-score

sd_age <- 12.94

mean_age <- mean(age)

z <- (value-mean_age)/sd_age

cat("Z-score =",z,"\n")

# Decimal Scaling

decimal <- value/100

cat("Decimal Scaling =",decimal)
