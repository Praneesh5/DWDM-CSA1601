# Program 9

marks <- c(55,60,71,63,55,65,50,55,
58,59,61,63,65,67,71,72,75)

marks <- sort(marks)

print(marks)

# Equal Frequency

bins <- split(marks,
ceiling(seq_along(marks)/(length(marks)/3)))

print(bins)

# Equal Width

breaks <- seq(min(marks),
max(marks),
length.out=4)

cut(marks,breaks=breaks,
include.lowest=TRUE)

hist(marks,
main="Histogram",
xlab="Marks",
col="lightblue")
